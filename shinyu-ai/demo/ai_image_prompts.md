## 実装例（Python + OpenAI API）

```python
import openai
import os

openai.api_key = os.getenv("OPENAI_API_KEY")

def generate_fortune_images(animal, color, vegetable, mushroom, weather):
    """占い結果から4枚の画像を生成"""

    prompts = {
        "animal": f"A majestic {animal.lower()} with {color.lower()} accents, sitting gracefully in a mystical setting. Digital art, fantasy illustration, 4K quality.",
        "vegetable": f"An anthropomorphic {vegetable.lower()} character with a cheerful personality. Cute kawaii illustration style.",
        "mushroom": f"A cluster of {mushroom.lower()} mushrooms with subtle elegance. Minimalist illustration style.",
        "weather": f"A beautiful {weather.lower()} weather element in a serene atmosphere. High detail, ethereal style."
    }

    images = {}

    for key, prompt in prompts.items():
        response = openai.images.generate(
            model="dall-e-3",
            prompt=prompt,
            size="1024x1024",
            quality="standard",
            n=1
        )

        images[key] = response.data[0].url

    return images

# 使用例
images = generate_fortune_images(
    animal="black panther",
    color="green",
    vegetable="tomato",
    mushroom="shimeji",
    weather="snow"
)

print("Generated Images:")
for key, url in images.items():
    print(f"{key}: {url}")
```
