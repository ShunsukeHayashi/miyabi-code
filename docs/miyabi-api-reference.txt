# Miyabi MCP API Reference
# Quick reference for all available MCP tools

================================================================================
MIYABI-GITHUB (GitHub Operations)
================================================================================

github_list_issues(state?, labels?, per_page?)
  List repository issues
  state: "open" | "closed" | "all" (default: "open")
  labels: comma-separated label names
  per_page: number (default: 30)

github_get_issue(issue_number)
  Get detailed issue information

github_create_issue(title, body?, labels?, assignees?)
  Create new issue

github_update_issue(issue_number, title?, body?, state?, labels?)
  Update existing issue

github_add_comment(issue_number, body)
  Add comment to issue or PR

github_list_prs(state?, per_page?)
  List pull requests

github_get_pr(pull_number)
  Get PR details

github_create_pr(title, head, base?, body?, draft?)
  Create pull request

github_merge_pr(pull_number, merge_method?)
  Merge PR (merge_method: "merge" | "squash" | "rebase")

github_list_labels()
  List repository labels

github_add_labels(issue_number, labels[])
  Add labels to issue/PR

github_list_milestones(state?)
  List milestones

dev_issue_create(title, body?, labels?, priority?)
  Create local DevIssue (offline queue)

dev_issue_list(status?)
  List local DevIssues

dev_issue_sync(owner?, repo?)
  Sync queued DevIssues to GitHub

obsidian_search(query, category?, tags?, case_sensitive?)

obsidian_get_tags()

obsidian_get_categories()

obsidian_get_directory_tree(base_path?)

obsidian_get_backlinks(path)

tmux_pane_search(pane_id, pattern)

tmux_pane_tail(pane_id, lines)

tmux_pane_is_busy(pane_id)

tmux_pane_current_command(pane_id)

tmux_join_commhub()

tmux_get_commhub_status()
